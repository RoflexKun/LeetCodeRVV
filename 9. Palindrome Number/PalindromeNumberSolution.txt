class Solution {
public:
    bool isPalindrome(int x) {
        if(x < 0)
            return false;
        else 
        {
            long long temp = x;
            long long ogl = 0;
            while(temp > 0)
            {
                ogl = ogl*10 + temp%10;
                temp/=10;
            }
            if(ogl == x)
                return true;
            else 
                return false;
        }
    }
};
